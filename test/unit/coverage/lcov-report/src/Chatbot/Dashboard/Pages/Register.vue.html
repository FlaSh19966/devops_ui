<script>
import swal from "sweetalert2";
import axios from "axios";
import api_calls from "@/api_calls";
export default {
  components: {},
  data() {
    return {
      user: {
        company_name: "",
        email: "",
        password: "",
        domain: "",
        services: "",
        host_name: "",
        registration_type: "Customer",
        license_key:
          Math.random()
            .toString(36)
            .substring(2, 15) +
          Math.random()
            .toString(36)
            .substring(2, 15),
        referral_id: "null"
      },
      companies: [],
      unique: false,
      input_disable: false,
      spinnerOn: false,
      reg_pwd: /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[#@!%*?&])[A-Za-z\d#@!%*?&]{8,}/,
      reg_email: /^([0-9a-zA-Z]([-.\w]*[0-9a-zA-Z])*@([0-9a-zA-Z][-\w]*[0-9a-zA-Z]\.)+[a-zA-Z]{2,9})$/,
      reg_referral: /company_id=([0-9a-zA-Z\s\+%20]+)/g,
      reg_token: /token=([0-9a-zA-Z._-]+)$/g,
      reg_invitee: /invitee=([0-9a-zA-Z]([-.\w]*[0-9a-zA-Z])*[@|%40]([0-9a-zA-Z][-\w]*[0-9a-zA-Z]\.)+[a-zA-Z]{2,9})/g
    };
  },
  computed: {
    validEmail() {
      if (this.reg_email.test(this.user.email)) {
        return true;
      } else {
        return false;
      }
    },
    valid_cname() {
      if (this.user.company_name != "") {
        return true;
      } else {
        return false;
      }
    },
    validPassword() {
      if (this.reg_pwd.test(this.user.password)) {
        return true;
      } else {
        return "Password must contain minimum 8 characters, atleast 1 UpperCase Alphabet, 1 LowerCase Alphabet, 1 Number and 1 Special Character";
      }
    }
    // service_types() {
    //   if (this.chatbot_check && this.voice_system_check) {
    //     (this.user.services = "Chatbot"), "Voice System";
    //   } else if (this.chatbot_check) {
    //     this.user.services = "Chatbot";
    //   } else if (this.voice_system_check) {
    //     this.user.services = "Voice System";
    //   } else {
    //     this.user.services = "";
    //   }
    // }
  },
  created() {
    axios.get(api_calls.domain_url()).then(response => {
      this.companies = [];
      for (var i in response.data) {
        this.companies.push(response.data[i]);
      }
    });
  },
  methods: {
    password_toggle(event) {
      event.preventDefault();
      var password = document.getElementById("form_password");
      if (password.type == "text") {
        password.type = "password";
      } else if (password.type == "password") {
        password.type = "text";
      }
    },
    toggle_domain(event) {
      if (event.target.value == "Customer") {
        document.getElementById("domain_input").style.display = "";
      } else if (event.target.value == "Partner") {
        document.getElementById("domain_input").style.display = "none";
      }
    },
    create(event) {
      event.preventDefault();
      if (
        this.validEmail &&
        this.validPassword == true &&
        this.valid_cname &&
        this.user.registration_type != null
      ) {
        this.call_api();
      } else if (!this.valid_cname) {
        swal({
          type: "error",
          position: "top-right",
          text: "Company Name is required",
          showConfirmButton: false,
          timer: 1000
        });
      } else if (!this.validEmail) {
        swal({
          type: "error",
          text: "Please enter a valid email",
          showConfirmButton: false,
          timer: 1000
        });
      } else if (this.user.password == "") {
        swal({
          type: "error",
          text: "Password field is blank",
          showConfirmButton: false,
          timer: 1000
        });
      } else if (this.user.registration_type == null) {
        this.$toastr.e("Please select the type of registration");
      }
    },
    call_api() {
      this.spinnerOn = true;
      axios
        .post(api_calls.register(), this.user)
        .then(response => {
          this.spinnerOn = false;
          if (response.data == "Registration Successful") {
            swal({
              title:
                "Account created successfully!!<br>Kindly check your e-mail for account validation",
              type: "success"
            });
          } else if (response.data.includes("This email is a duplicate.")) {
            swal({
              type: "error",
              text:
                "This Email is already registered. Please try with another email"
            });
          } else if (response.data.includes("Registration Unsuccessful")) {
            swal({
              type: "error",
              text: "Your registration was unsuccessful. Please try again."
            });
          } else if (response.data == "Internal Server Error") {
            swal({
              type: "error",
              text: "There was an internal server error. Please try again."
            });
          } else {
            swal({
              type: "error",
              text: "Cannot register your account",
              showConfirmButton: false,
              timer: 1000
            });
          }
          this.user.email = "";
          this.user.company_name = "";
          this.user.password = "";
          this.user.domain = "";
        })
        .catch(e => {
          this.spinnerOn = false;
        });
    }
  },
  beforeMount() {
    var link = window.location.href;
    var referral_id = String(link.match(this.reg_referral)).replace(
      "company_id=",
      ""
    );
    var invitee = String(link.match(this.reg_invitee)).replace("invitee=", "");
    var token = String(link.match(this.reg_token)).replace("token=", "");
    invitee = invitee.replace(/%40/, "@");
    if (referral_id != "null" && invitee != "null" && token != "null") {
      this.user.referral_id = referral_id;
      this.user.registration_type = "Customer";
      this.user.email = invitee;
      this.input_disable = true;
    }
  }
};
</script>
