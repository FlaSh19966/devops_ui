<template>
<div>
<div class="tab" style="margin-top:1%">
  <button @click="rend(1)" :style="first_element_style">Train the Bot</button>
  <button @click="rend(2)" :style="second_element_style">Response Bot</button>
</div>
<div class="tabcontent">
  <component :is="render_component" style="margin-top:1%"></component>
</div>
</div>
</template>
 
 
<script>
import Chat from '@/pages/Dashboard/Chat.vue'
import BotResponse from '@/pages/Dashboard/BotResponse.vue'
 
export default { 
  data () {
   return {
    render_component: 'chat',
    components_list: ['chat', 'bot-response'],
}
},
computed: {
  first_element_style () {
    if(this.render_component=='chat') {
      return 'background-color: #fff;'
    }
  },
  second_element_style () {
    if(this.render_component=='bot-response') {
      return 'background-color: #fff'
    }
  }
},
components: {
  Chat,
  BotResponse
},
beforeCreate () {
    if(!this.$session.exists()) {
      this.$router.push('/')
    }
  },
methods: {
  rend (index) {
    this.render_component = this.components_list[index-1]
  }
}
}
</script>
