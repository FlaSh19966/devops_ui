  <script>
  import axios from 'axios'
  import swal from 'sweetalert2'
 
  export default {
      name: 'config-api',
      data() {
          return {
            open_modal: false,
            post_link: process.env.CONFIG_API_URL,
            get_list_link: process.env.CONFIGURING_API_LIST,
            show_url_link: process.env.SHOW_URL_LIST,
            user_role: this.$session.get('UserInformation').role,
            url: [],
            name: null,
            show: false,
            show_urls: [],
            companyname: null,
            companyid: null,
            index: null,
            show_json_request_data: null,
            show_json_response_data: null,
            json_request_data: [],
            json_response_data: [],
            json_request_class: null,
            json_response_class: null,
            editable: false,
            type: null,
          }
      },
      created () {
  if(this.user_role=="SuperAdmin") {
    this.companyid = this.$session.get('CompanyId')
    this.companyname = this.$session.get('CompanyName')
  }
  else if(this.user_role == "Admin") {
    this.companyid = this.$session.get('UserInformation').company_id
    this.companyname = this.$session.get('UserInformation').company_name
  }
  if(this.user_role=='Admin') {
    axios.post(this.get_list_link, {
          companyid: this.companyid,
          companyname: this.companyname,
          // role: this.user_role,
          email: this.$session.get('UserInformation').email,
          licensekey: this.$session.get('UserInformation').license_key,
          token: this.$session.get('UserInformation').tokens
        }).then(response => {
          if(response.data == "already added") {
            this.show = this.show
          }
          else {
          this.bind_list(response.data)
          this.show =!(this.show)
          }
        })
  }
  else if(this.user_role=='SuperAdmin') {
  if(!this.$session.has('CompanyId')) {
     this.$toastr.w("Please select a company name from the dropdown")
   }
   else {
    axios.post(this.get_list_link, {
          companyid: this.companyid,
          companyname: this.companyname,
          // role: this.user_role,
          // superadmincompanyname: this.$session.get('UserInformation').company_name,
          // superadmincompanyid: this.$session.get('UserInformation').company_id,
          email: this.$session.get('UserInformation').email,
          licensekey: this.$session.get('UserInformation').license_key,
          token: this.$session.get('UserInformation').tokens
        }).then(response => {
          if(response.data == "already added") {
            this.show = this.show
          }
          else {
          this.bind_list(response.data)
          this.show =!(this.show)
          }
        })
   }
  }
        
        if(this.user_role=='Admin') {
          axios.post(this.show_url_link, {
          companyid: this.companyid,
          companyname: this.companyname,
          // superadmincompanyname: this.$session.get('UserInformation').company_name,
          // superadmincompanyid: this.$session.get('UserInformation').company_id,
          // role: this.user_role,
          email: this.$session.get('UserInformation').email,
          licensekey: this.$session.get('UserInformation').license_key,
          token: this.$session.get('UserInformation').tokens
        }).then(response => {
          this.show_urls_update(response.data[0],response.data[1],response.data[2])
        })
        }
        else if(this.user_role=='SuperAdmin') {
          if(!this.$session.has('CompanyId')) {
   }
   else {
          axios.post(this.show_url_link, {
          // superadmincompanyname: this.$session.get('UserInformation').company_name,
          // superadmincompanyid: this.$session.get('UserInformation').company_id,
          companyid: this.companyid,
          companyname: this.companyname,
          // role: this.user_role,
          email: this.$session.get('UserInformation').email,
          licensekey: this.$session.get('UserInformation').license_key,
          token: this.$session.get('UserInformation').tokens
        }).then(response => {
          this.show_urls_update(response.data[0],response.data[1], response.data[2])
        })
        }
        }
      },
      methods: {
        modal_open (index, type) {
          this.index = index
          this.type = type
          if(type=='edit') {
            axios.post(this.show_url_link, {
              companyid: this.companyid,
              companyname: this.companyname,
              // superadmincompanyname: this.$session.get('UserInformation').company_name,
              // superadmincompanyid: this.$session.get('UserInformation').company_id,
              // role: this.user_role,
              email: this.$session.get('UserInformation').email,
              question: this.show_urls[index].question,
              response: this.show_urls[index].response,
              popupjson: "True",
              licensekey: this.$session.get('UserInformation').license_key,
              token: this.$session.get('UserInformation').tokens
            }).then(response => {
              this.show_json_request_data = response.data[0]
              this.show_json_response_data = response.data[1]
            })
          }
          this.json_request_class = null
          this.json_response_class = null
          this.open_modal = true
        },
        modal_close () {
          if(this.json_request_data[this.index]=="" || this.json_response_data[this.index]=="" 
    || this.json_request_class=="md-error" || this.json_response_class=="md-error") {
      this.json_request_data[this.index] = ''
      this.json_response_data[this.index] = ''
    }
    this.show_json_request_data = null
    this.show_json_response_data = null
          this.open_modal = false
          this.index = null
        },
        show_urls_update(message1, message2, message3) {
          if(message1==null || message2==null) {
          }
          else {
            for(var i in message1) {
              this.show_urls.push({
                question: message1[i],
                response: message2[i],
                data: message3[i]
              })
            }
          }
        },
        bind_list(message) {
          this.name = message
          for(var i in this.name) {
            this.json_request_data.push('')
            this.json_response_data.push('')
            this.url[i] = ""
          }
        },
  add_url () {
      if(this.url.includes("")){
        this.$toastr.e("Kindly enter the blank fields")
      }
      else if(this.json_request_data.includes("") || this.json_request_data.includes("")) {
        this.$toastr.e("Kindly enter valid JSON data")
      }
      else {
    axios.post(this.post_link, {
      companyname: this.companyname,
      companyid: this.companyid,
      questionlist: this.name,
      urllist: this.url,
      email: this.$session.get('UserInformation').email,
      requestjson: this.json_request_data,
      responsejson: this.json_response_data,
      licensekey: this.$session.get('UserInformation').license_key,
      token: this.$session.get('UserInformation').tokens
    }).then(response => { 
      if(response.data=="url added") {
      swal({
        type: 'success',
        text: "All URLs added successfully",
      })
      this.$router.go()
    } });
    this.url=[];
      }
  },
  json_request_validate () {
    try {
      if(this.type=='create') {
      JSON.parse(this.json_request_data[this.index])
    }
    else if(this.type=='edit' && this.editable) {
      JSON.parse(this.show_json_request_data)
    }
    }
    catch (e) {
      this.json_request_class = 'md-error'
      return
    }
    this.json_request_class = 'md-valid'
    return
  },
  json_response_validate () {
    try {
      if(this.type=='create') {
      JSON.parse(this.json_response_data[this.index])
    }
    else if(this.type=='edit' && this.editable) {
      JSON.parse(this.show_json_response_data)
    }
    }
    catch (e) {
      this.json_response_class = 'md-error'
      return
    }
    this.json_response_class = 'md-valid'
    return
  },
  submit_json () {
    if(this.type=='create') {
    if(this.json_request_data[this.index]=="" || this.json_response_data[this.index]=="" 
    || this.json_request_class=="md-error" || this.json_response_class=="md-error") {
      this.$toastr.e("Kindly enter all the required valid fields")
    }
    else if(this.json_request_class!="md-error" && this.json_response_class!="md-error") {
      this.open_modal = false
    }
  }
  else if(this.type=='edit' && this.editable) {
    axios.post(this.post_link, {
      companyname: this.companyname,
      companyid: this.companyid,
      // superadmincompanyname: this.$session.get('UserInformation').company_name,
      // superadmincompanyid: this.$session.get('UserInformation').company_id,
      // role: this.user_role,
      email: this.$session.get('UserInformation').email,
      questionlist: this.show_urls[this.index].question,
      urllist: this.show_urls[this.index].response,
      requestjson: this.show_json_request_data,
      responsejson: this.show_json_response_data,
      isedit: "True",
      licensekey: this.$session.get('UserInformation').license_key,
      token: this.$session.get('UserInformation').tokens
    }).then(response => {
      if(response.data=="json updated") {
      this.$toastr.s("Request and Response JSON updated successfully")
      this.show_json_request_data = null
      this.show_json_response_data = null
      this.editable = false
      this.open_modal = false
      }
    })
  }
  }
}
}
