<script>
import axios from 'axios'
import swal from 'sweetalert2'
export default {
    name: 'email',
    data() {
      return {
      read: false,
      unanswered_url: process.env.GET_UNANSWERED_URL,
      template_name: '',
      send_url: process.env.EMAIL_SEND_URL,
      show_template_url: process.env.SHOW_TEMPLATE_URL,
      list: [],
      sender_list: '',
      subject: '',
      email_body: '',
      companyname: null,
      companyid: null,
      user_role: this.$session.get('UserInformation').role,
      isEdit: "False",
      intent_list: [],
      intent: '',
      showProgress: false
    }
    },
    created () {
    if(this.user_role=="SuperAdmin") {
    if(this.$session.has('CompanyId')) {
    this.companyid = this.$session.get('CompanyId')
    this.companyname = this.$session.get('CompanyName')
  }
  else {
    this.companyid = this.$session.get('UserInformation').company_id
    this.companyname = this.$session.get('UserInformation').company_name
  }
  }
  else if(this.user_role == "Admin") {
    this.companyid = this.$session.get('UserInformation').company_id
    this.companyname = this.$session.get('UserInformation').company_name
  }
  if(this.user_role=='Admin') {
    axios.post(this.show_template_url, {
    companyname: this.companyname,
    companyid: this.companyid,
    isEdit: "False",
    // role: this.user_role,
    email: this.$session.get('UserInformation').email,
    licensekey: this.$session.get('UserInformation').license_key,
    token: this.$session.get('UserInformation').tokens
  }).then(response => {
    for(var i in response.data[0]){
     this.list.push({template_name: response.data[0][i], intent: response.data[1][i]});
    }
  })
  }
  else if(this.user_role=='SuperAdmin') {
  if(!this.$session.has('CompanyId') && this.$session.get('UserInformation').role=='SuperAdmin') {
     this.$toastr.w("Please select a company name from the dropdown")
   }
   else {
    axios.post(this.show_template_url, {
    companyname: this.companyname,
    companyid: this.companyid,
    // superadmincompanyname: this.$session.get('UserInformation').company_name,
    // superadmincompanyid: this.$session.get('UserInformation').company_id,
    email: this.$session.get('UserInformation').email,
    isEdit: "False",
    // role: this.user_role,
    licensekey: this.$session.get('UserInformation').license_key,
    token: this.$session.get('UserInformation').tokens
  }).then(response => {
    for(var i in response.data[0]){
     this.list.push({template_name: response.data[0][i], intent: response.data[1][i]});
    }
  })
   }
  }
  axios.post(this.unanswered_url, {
            companyid: this.companyid,
            companyname: this.companyname,
            isDropdown: "True",
            // role: this.$session.get('UserInformation').role,
            // superadmincompanyname: this.$session.get('UserInformation').company_name,
            // superadmincompanyid: this.$session.get('UserInformation').company_id,
            email: this.$session.get('UserInformation').email,
            licensekey: this.$session.get('UserInformation').license_key,
            token: this.$session.get('UserInformation').tokens
        })
        .then(response => {
           this.intent_list = response.data
        })
    },
    methods: {
      clear_fields () {
        this.sender_list = ""
        this.template_name = ""
        this.subject = ""
        this.email_body = ""
        this.read = false
        this.isEdit = "False"
        this.intent = ''
      },
      add_mail() {
        var edit = "this.read==true"?"True":"False"
        if(this.template_name=="" || this.sender_list=="" || this.email_body=="" || this.subject=="") {
          this.$toastr.e("Please enter the required fields")
        }
        else {
        axios.post(this.send_url, {
          companyname: this.companyname,
          companyid: this.companyid,
          // superadmincompanyname: this.$session.get('UserInformation').company_name,
          // superadmincompanyid: this.$session.get('UserInformation').company_id,
          email: this.$session.get('UserInformation').email,
          senderlist: this.sender_list,
          subject: this.subject,
          templatename: this.template_name,
          emailbody: this.email_body,
          isEdit: this.isEdit,
          // role: this.user_role,
          intent: this.intent,
          licensekey: this.$session.get('UserInformation').license_key,
          token: this.$session.get('UserInformation').tokens
        }).then(response => {
          if(response.data=="template successfully") {
            swal({
              text: 'Email template has been added successfully'
            })
            // setTimeout( ()=> {
            //   this.showProgress = true
            // }, 500)
          }
          else if (response.data=="Emailtemplate updated") {
            this.$notify(
        {
          message: 'Your email template has been successfully updated',
          icon: 'add_alert',
          horizontalAlign: 'right',
          verticalAlign: 'top',
          type: 'success'
        })
        // setTimeout( ()=> {
        //       this.showProgress = true
        //     }, 500)
          }
          else if(response.data.includes("E11000 duplicate key error")) {
            swal({
              type: 'warning',
              text: 'Template already exists. Please try with another template name'
            })
          }
          else {
            swal({
              type: 'error',
              text: 'Your template has not been added'
            })
          }
        })
        }
        this.isEdit = "False"
        this.sender_list = ""
        this.template_name = ""
        this.subject = ""
        this.email_body = ""
        this.read = false
        this.intent = ''
        this.showProgress = false
      },
      edit_mail (index) {
        this.isEdit = "True"
  if(!this.$session.has('CompanyId') && this.$session.get('UserInformation').role=='SuperAdmin') {
     this.$toastr.w("Please select a company name from the dropdown")
   }
  else {
    axios.post(this.show_template_url, {
    companyname: this.companyname,
    companyid: this.companyid,
    isEdit: this.isEdit,
    email: this.$session.get('UserInformation').email,
    // superadmincompanyname: this.$session.get('UserInformation').company_name,
    // superadmincompanyid: this.$session.get('UserInformation').company_id,
    licensekey: this.$session.get('UserInformation').license_key,
    token: this.$session.get('UserInformation').tokens,
    // role: this.user_role,
    templatename: this.list[index].template_name
  }).then(response => {
    this.template_name = response.data.templatename
    this.subject = response.data.subject
    this.email_body = response.data.body
    this.intent = response.data.intent
    this.sender_list = response.data.sender
    this.read = true
  })
        }
      }
    }
}
</script>
