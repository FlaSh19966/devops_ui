    <div>
        <transition name="modal" v-if="open_modal">

<div class="modal-body text-center">
    <div class="md-layout">
        <div class="md-layout-item md-size=100">
    <md-field>
        <label>Enter your message here....</label>
        <md-textarea v-model="message"></md-textarea>
    </md-field>
        </div>
        <div class="md-layout-item md-size-100">
<md-button native-type="submit" class="md-success" @click="send_mail">Send Email</md-button>

    </div>
      </div>
    </div>
      </div>

        <div class="md-layout">
            <md-card>
                <md-card-content>
                    <div class="md-layout"> 
                        <div class="md-layout-item md-size-25">
                            <h5 style="font-weight:bold">Company Name</h5>
                        </div>

                    <div class="md-layout" v-for="(item, index) in client_data" :key="index">
                        <div class="md-layout-item md-size-25">
                            <p>{{ item.companyname }}</p>
                        </div>
                        <div class="md-layout-item md-size-30">
                            <p>{{ item.email }}</p>
                        </div>
                        <div class="md-layout-item md-size-20">
                            <p>{{ item.status }}</p>
                        </div>
                        <div class="md-layout-item md-size-25">
                <md-button class="md-success md-sm" v-if="item.status=='Pending' || item.status=='Decline'"
                style="margin-top:5%" @click="modal_open(item, 'Accept')">Accept</md-button>
                <md-button class="md-danger md-sm" v-if="item.status=='Pending' || item.status=='Accept'" 
                style="margin-top:5%" @click="modal_open(item, 'Decline')">Decline</md-button>
                        </div>

<script>
import axios from 'axios'
export default {
    data () {
        return {
            client_data: [],
            user_role: this.$session.get('UserInformation').role,
            clients_list_url: process.env.CLIENTS_APPROVAL_URL,
            open_modal: false,
            message: null,
            client_name: null,
            type: null
        }
    },
    beforeCreate () {
    if(!this.$session.exists()) {
      this.$router.push('/')
    }
    },
    created () {
        axios.post(this.clients_list_url, {
            // superadmincompanyname: this.$session.get('UserInformation').company_name,
            // superadmincompanyid: this.$session.get('UserInformation').company_id,
            email: this.$session.get('UserInformation').email,
            token: this.$session.get('UserInformation').tokens,
            licensekey: this.$session.get('UserInformation').license_key,
            showlist: "True"})
            .then(response => {
            for(var i in response.data[0]) {
                this.client_data.push({
                    companyname: response.data[0][i],
                    email: response.data[1][i],
                    status: response.data[2][i],
                    companyid: response.data[3][i]
                })
            }
        })
    },
    methods: {
        modal_open (item, type) {
            this.client_name = item
            this.type = type
            if(type=='Accept') {
                this.send_mail()
                this.open_modal = false
                return
            }
            else if(type=='Decline') {
            this.open_modal = true
            return
            }
        },
        modal_close () {
            this.open_modal = false
        },
        send_mail () {
            axios.post(this.clients_list_url, {
                companyname: this.client_name.companyname,
                companyid: this.client_name.companyid,
                // superadmincompanyname: this.$session.get('UserInformation').company_name,
                // superadmincompanyid: this.$session.get('UserInformation').company_id,
                acceptdeclinebit: "True",
                acceptdecline: this.type,
                message: this.message,
                email: this.$session.get('UserInformation').email,
                clientemail: this.client_name.email,
                token: this.$session.get('UserInformation').tokens,
                licensekey: this.$session.get('UserInformation').license_key
            }).then(response => {
                this.$toastr.i(response.data)
                this.message = null
                this.open_modal = false
                this.client_data.status = this.type
                this.$router.go()
            })
        }
    }
}
</script>
