<script>
import swal from 'sweetalert2'
import axios from 'axios'
 
export default {
  name: 'extended-forms',
  data () {
    return {
      show_superadmin_data: [],
      add_superadmin_data: [],
      questionlist: [],
      responselist: null,
      intentlist: [],
      sourcelist: null,
      user_options: ["Mobile"],
      options: [],
      user_role: this.$session.get('UserInformation').role,
      add: 0,
      questions: [''],
      response: [''],
      companyname: null,
      companyid: null,
      add_question_url: process.env.ADD_QUESTION_URL,
      show_question_url: process.env.SHOW_QUESTION_URL,
      options_list_url: process.env.SOURCE_LIST_URL,
      load_admin_url: process.env.LOAD_ADMIN_QUESTIONS_URL,
      add_intent_url: process.env.SUBMIT_INTENT_URL,
}
  },
  computed: {
    questions_class () {
      if(this.user_role=='SuperAdmin') {
        return 'md-layout-item md-size-25'
      }
      else {
        return 'md-layout-item md-size-25'
      }
    },
    response_class () {
      if(this.user_role=='SuperAdmin') {
        return 'md-layout-item md-size-40'
      }
      else {
        return 'md-layout-item md-size-65'
      }
    },
    sources_class () {
      if(this.user_role=='SuperAdmin') {
        return 'md-layout-item md-size-10'
      }
      else {
        return 'md-layout-item md-size-10'
      }
    }
  },
  beforeCreate () {
    if(!this.$session.exists()) {
      this.$router.push('/')
    }
  },
  created () {
    if(this.user_role=="SuperAdmin") {
    if(this.$session.has('CompanyId')) {
    this.companyid = this.$session.get('CompanyId')
    this.companyname = this.$session.get('CompanyName')
  }
  else {
    this.companyid = this.$session.get('UserInformation').company_id
    this.companyname = this.$session.get('UserInformation').company_name
  }
  }
  else if(this.user_role == "Admin") {
    this.companyid = this.$session.get('UserInformation').company_id
    this.companyname = this.$session.get('UserInformation').company_name
  }
      axios.post(this.load_admin_url, {
      role: this.user_role,
      companyname: this.companyname,
      // superadmincompanyname: this.$session.get('UserInformation').company_name,
      // superadmincompanyid: this.$session.get('UserInformation').company_id,
      email: this.$session.get('UserInformation').email,
      companyid: this.companyid,
      licensekey: this.$session.get('UserInformation').license_key,
      token: this.$session.get('UserInformation').tokens
    }).then(response => {
      this.update_qns(response.data)
  })
 
  axios.get(this.options_list_url).then(response => {
         this.options = response.data
       })
  },
  methods: {
    add_intent () {
      if(this.add_superadmin_data.intent.includes("")) {
        swal({
          type: 'warning',
          text: 'Please enter all the intents'
        })
      }
      else if(!this.$session.has('CompanyId') && this.$session.get('UserInformation').role=='SuperAdmin') {
     this.$toastr.w("Please select a company name from the dropdown")
   }
      else {
      for(var i in this.add_superadmin_data) {
        this.intentlist.push(this.add_superadmin_data[i].intent)
      }
      axios.post(this.add_intent_url, {
      companyname: this.companyname,
      companyid: this.companyid,
      questionlist: this.questionlist,
      responselist: this.responselist,
      intentlist: this.intentlist,
      sourcelist: this.sourcelist,
      // superadmincompanyname: this.$session.get('UserInformation').company_name,
      // superadmincompanyid: this.$session.get('UserInformation').company_id,
      // role: this.user_role,
      email: this.$session.get('UserInformation').email,
      licensekey: this.$session.get('UserInformation').license_key,
      token: this.$session.get('UserInformation').tokens
      }).then(response => {
        if(response.data=="intent added successfully") {
        swal({
          text: "Intents added successfully",
          type: 'success'
        })
        }
      })
      this.add_superadmin_data = []
      }
    },
    update_qns(message){
      var questionlist, responselist, sourcelist
      var show_questionlist, show_responselist, show_sourcelist, show_intentlist
 
        if(this.user_role=="Admin") {
      questionlist = message.questionlist
      responselist = message.responselist
      sourcelist = message.sourcelist
        for(var i in questionlist) {
          this.add_superadmin_data.push({
            question: questionlist[i],
            response: responselist[i],
            source: sourcelist[i],
            intent: ""
          })
        }
        }
        else if(this.user_role=="SuperAdmin") {
 
      questionlist = message.add_intent_field.questionlist
      responselist = message.add_intent_field.responselist
      sourcelist = message.add_intent_field.sourcelist
      show_questionlist = message.show_intent_field.intent_questionlist
      show_responselist = message.show_intent_field.intent_responselist
      show_sourcelist = message.show_intent_field.intent_sourcelist
      show_intentlist = message.show_intent_field.intent_intentlist
 
      for(var i in questionlist) {
          this.add_superadmin_data.push({
            question: questionlist[i],
            response: responselist[i],
            source: sourcelist[i],
            intent: ""
          })
        }
        for(var i in show_questionlist) {
           this.show_superadmin_data.push({
             question: show_questionlist[i],
             response: show_responselist[i],
             source: show_sourcelist[i],
             intent: show_intentlist[i]
           })
         }
        }
        this.questionlist = questionlist
        this.responselist = responselist
        this.sourcelist = sourcelist
    },
    check (message) {
      if(message=='add'){
      this.add += 1 ;
      this.questions.push('');
      this.response.push('');
      this.user_options.push('Mobile');
      }
      else if(message=='remove'){
        this.add -= 1;
        this.questions.pop('');
        this.response.pop('');
        this.user_options.pop();
      }
    },
    add_qn () {   
        if(this.questions.includes("")==true){
          this.$toastr.w("Please enter a valid question")
      //     swal(
      // { 
      //   position: 'top-right',
      //   text: 'Please Enter a Valid Question',
      //   customClass: 'swal-size',
      // });
       }
      else if(this.response.includes("")==true){
        this.$toastr.w("Please enter a valid response")
      //   swal(
      // { 
      //   position: 'top-right',
      //   text: 'Please Enter a Valid Response',
      //   customClass: 'swal-size',
      // });
      }
      else if(this.user_options.includes("")==true){
        this.$toastr.w("Please select a valid option")
      //   swal(
      // { 
      //   position: 'top-right',
      //   text: 'Please Enter a Valid Response',
      //   customClass: 'swal-size',
      // });
      }
       else {
      axios.post(this.add_question_url, {
       questions: this.questions, 
       response: this.response,
       source_list: this.user_options,
       company_name: this.companyname,
       company_id: this.companyid,
       email: this.$session.get('UserInformation').email,
       licensekey: this.$session.get('UserInformation').license_key,
       token: this.$session.get('UserInformation').tokens
       }).then(response => {
         var response_api = response.data;
         if(response_api=="added_successfully"){
        swal(
      { 
        text: 'All your questions and responses have been added successfully!!',
        type: "success"
      });
         }
      });
      this.questions = [""];
      this.response = [""];
      this.user_options = [""];
      this.add = 0;
      }
    },
    append (message) {
      for(var i in message.question_list)
      {
        this.received.push ({
          question: message.question_list[i], 
          response: message.response_list[i]
          })
      }
       this.appendkey=!(this.appendkey);
     },
  },
 }
</script>
