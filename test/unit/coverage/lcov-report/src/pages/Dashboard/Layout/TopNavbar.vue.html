<script>
import ChangePassword from '@/pages/Dashboard/Pages/ChangePassword.vue'
import axios from 'axios'
export default {
  data () {
    return {
      company_selected: this.$session.get('CompanyName'),
      company_list_url: process.env.COMPANY_LIST_URL,
      signout_url: process.env.SIGNOUT_URL,
      companies: [],
      company_id: [],
      user_company: null,
      user_company_id: this.$session.get('UserInformation').company_id,
      user_role: this.$session.get('UserInformation').role,
      open_modal: false
    }
  },
  components: {
    ChangePassword
  },
  watch: {
    company_selected (newVal, oldVal) {
      if(newVal=='Select Company Name') {
 
      }
      else {
      if(this.$session.has('CompanyName')){
        this.$session.remove('CompanyName')
      }
      this.$session.set('CompanyName',newVal)
    if(this.$session.has('CompanyId')){
        this.$session.remove('CompanyId')
      }
      this.$session.set('CompanyId',this.company_id[this.companies.indexOf(newVal)])
      this.$router.go()
      return newVal
    }
    }
  },
  created () {
    if(this.user_role=='Admin') {
      this.company_selected = this.$session.get('UserInformation').company_name
    }
  },
  methods: {
    company_list_api() {
      axios.post(this.company_list_url, {
        companyname: this.$session.get('UserInformation').company_name,
        companyid: this.$session.get('UserInformation').company_id,
        licensekey: this.$session.get('UserInformation').license_key,
        email: this.$session.get('UserInformation').email,
        token: this.$session.get('UserInformation').tokens
      }).then(response => {
        this.companies = response.data[0]
        this.company_id = response.data[1]
      })
    },
    toggleSidebar () {
      this.$sidebar.displaySidebar(!this.$sidebar.showSidebar)
    },
    minimizeSidebar () {
      if (this.$sidebar) {
        this.$sidebar.toggleMinimize()
      }
    },
    change_password () {
      this.$router.push('/change-password')
    },
    signout_api () {
      axios.post(this.signout_url, {
        companyname: this.$session.get('UserInformation').company_name,
        companyid: this.$session.get('UserInformation').company_id,
        licensekey: this.$session.get('UserInformation').license_key,
        email: this.$session.get('UserInformation').email,
        token: this.$session.get('UserInformation').tokens
      }).then(response => {
        if(response.data=='sign out') {
       this.$session.destroy()
       this.$router.push('/login')
        }
        else {
 
        }
      })
    }
  },
  
}
</script>
