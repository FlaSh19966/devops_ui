<template>
    <div style="margin-top:3%">
        <md-tabs>
             <md-tab v-for="(routes, index) in menu_list" :key="index" 
             :md-label="routes.text" @click="write(routes)">
             </md-tab>
        </md-tabs>
          <zoom-center-transition :duration="200" mode="out-in">
            <component :is="render_component"></component>
        </zoom-center-transition>
</div>
</template>
 
<script>
import Unanswered from '@/pages/Dashboard/Unanswered.vue'
import TrainBot from '@/pages/Dashboard/TrainBot.vue'
import {ZoomCenterTransition} from 'vue2-transitions'
import axios from 'axios'
import links from './links'
 
export default {
    components: {
    ZoomCenterTransition,
    Unanswered,
    TrainBot
  },
  data () {
  return {
    menu_list: [],
    links: links,
    menu_list_url: process.env.MENU_LIST_URL,
    render_component: 'unanswered',
}
},
beforeCreate () {
    if(!this.$session.exists()) {
      this.$router.push('/')
    }
  },
created () {
  axios.post(this.menu_list_url, {
    role: this.$session.get('UserInformation').role,
    companyid: this.$session.get('UserInformation').company_id,
    companyname: this.$session.get('UserInformation').company_name,
    licensekey: this.$session.get('UserInformation').license_key,
    token: this.$session.get('UserInformation').tokens,
    superadmincompanyname: this.$session.get('UserInformation').company_name,
    superadmincompanyid: this.$session.get('UserInformation').company_id,
    menuname: "Train"
  }).then(response => {
    var data = response.data
    for (var i in this.links){
      if(data.includes(this.links[i].text)){
        this.menu_list.push(this.links[i])
      }
    }
  })
},
methods: {
  write(route) {
    this.render_component = route.page
  }
}
}
</script>
