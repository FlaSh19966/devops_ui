          token: '',

          email: ''

}
},

],

    if(this.$session.has('CompanyId')) {
    this.companyid = this.$session.get('CompanyId')
    this.companyname = this.$session.get('CompanyName')
  }
  else {
    this.companyid = this.$session.get('UserInformation').company_id
    this.companyname = this.$session.get('UserInformation').company_name
  }
  }
  if(!this.$session.has('CompanyId') && this.$session.get('UserInformation').role=='SuperAdmin') {
     this.$toastr.w("Please select a company name from the dropdown")
   }
   else {
    if(this.$session.has('ChatMessage'+this.companyname)) {
      var chat = this.$session.get('ChatMessage'+this.companyname)
      for(var i in chat) {
        this.meta.push(chat[i])
      }
    }
   }
   axios.post(this.unanswered_url, {
            companyid: this.companyid,
            companyname: this.companyname,
            isDropdown: "True",
            // role: this.$session.get('UserInformation').role,

  updated () {
this.$refs.scroll.$el.scrollTop = this.$refs.scroll.$el.scrollHeight
  },
methods: {
 edit_submit () {
   if(this.intent == "Select Intent" || this.intent_value == ""){
     this.$toastr.e("Please Enter the values")
   }
   else if(!this.$session.has('CompanyId') && this.$session.get('UserInformation').role=='SuperAdmin') {
     this.$toastr.w("Please select a company name from the dropdown")
   }
   else {
    axios.post(this.edit_intent_url, {
       companyname: this.companyname,
       companyid: this.companyid,
       question: this.meta[this.index].msg,
       editintent: this.intent,
       intentvalue: this.intent_value,
      //  role: this.$session.get('UserInformation').role,

      response => {

    this.meta[this.index].intent = this.intent

    this.intent = "Select Intent"

    this.open_modal = false
      this.$session.set(chat_name,this.meta)
        }

    this.open_modal = true

				if(this.messageToSend == ''){
					return false;
				}

        this.info.question = this.messageToSend

        this.info.companyname = this.companyname

                subject = response.data[1]
                email_body = response.data[2]
                this.send(sender, subject, email_body)
        })
        },
        send (sender, subject, email_body) {
        axios.post(this.notify_superadmin_url, {
                companyname: this.companyname,
                companyid: this.companyid,
                email: sender,
                body: email_body,
                subject: subject,
                bindemail: "False",

                licensekey: this.$session.get('UserInformation').license_key,
                token: this.$session.get('UserInformation').tokens
            }).then(response => {
                this.$toastr.s(response.data)
            })
        }
        }
}
