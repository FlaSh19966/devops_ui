        </md-card-content>
      </md-card>
<div class="md-layout-item md-size-100 md-small-size-100">
        <md-button v-if="companyname!==undefined" native-type="submit" class="md-success" @click="save_role">Save</md-button>
</div>

<script>
import { SignupCard } from "@/components"
import axios from "axios"
import swal from 'sweetalert2'
export default {
  components: {
    SignupCard
  },
 
  name: "permissions",
 
  data() {
    return {
      role_url: process.env.ROLE_URL,
      role_submit_url: process.env.ROLE_SUBMIT_URL,
      user_role: this.$session.get('UserInformation').role,
      user: [],
      roles: [],
      role_list: [],
      restriction_list: [],
      companyid: "this.$session.get('UserInformation').role=='SuperAdmin'"?this.$session.get('CompanyId')
      :this.$session.get('UserInformation').company_id,
 
      companyname: "this.$session.get('UserInformation').role=='SuperAdmin'"?this.$session.get('CompanyName')
      :this.$session.get('UserInformation').company_name
    };
  },
  props: {
    placeholder: [String]
  },
created () {
  if(this.user_role == "Admin") {
    this.companyid = this.$session.get('UserInformation').company_id
    this.companyname = this.$session.get('UserInformation').company_name
  }
  if(this.companyname!==undefined) {
      axios
        .post(this.role_url, {
          companyid: this.companyid,
          companyname: this.companyname,
          // superadmincompanyname: this.$session.get('UserInformation').company_name,
          // superadmincompanyid: this.$session.get('UserInformation').company_id,
          // role: this.user_role,
          email: this.$session.get('UserInformation').email,
          licensekey: this.$session.get('UserInformation').license_key,
          token: this.$session.get('UserInformation').tokens
        })
        .then(response => {
          this.update_list(response.data);
        })
  }
},
  methods: {
    save_role () {
      for(var i in this.roles) {
        this.role_list.push(this.roles[i].role)
        this.restriction_list.push(this.roles[i].user)
      }
      axios.post(this.role_submit_url, {
        companyid: this.companyid,
        companyname: this.companyname,
        rolename: this.role_list,
        // superadmincompanyname: this.$session.get('UserInformation').company_name,
        // superadmincompanyid: this.$session.get('UserInformation').company_id,
        // role: this.user_role,
        email: this.$session.get('UserInformation').email,
        restrictionlist: this.restriction_list,
        licensekey: this.$session.get('UserInformation').license_key,
        token: this.$session.get('UserInformation').tokens
      }).then( response => {
        if(response.data=="Add successfully") {
          swal({
            text: "Roles and restrictions updated successfully"
          })
          for(var i in this.roles) {
            if(this.roles[i].input_enable==true) {
              this.roles[i].input_enable=false
            }
          }
        }
      })
    },
    edit_restrictions (index) {
      if(this.roles[index].user!='') {
      this.roles[index].input_enable = !this.roles[index].input_enable
      }
    },
    update_list(response) {
      var input
      for (var i in response.restrictionlist) {
        if(response.restrictionlist[i]=='') {
          input = true
        }
        else {
         input = false
        }
          this.roles.push({
            user: response.restrictionlist[i],
            role: response.rolelist[i],
            input_enable: input
          });
      }
    },
  }
};
</script>
