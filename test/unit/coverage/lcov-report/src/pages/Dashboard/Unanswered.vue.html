<template>
    <div class="md-layout">
        <md-card>
            <md-card-header v-if="!unanswered_list.length>0">
                <h4 style="font-weight:bold">No unanswered questions left</h4>
            </md-card-header>
            <md-card-content v-if="unanswered_list.length>0 && companyname!==undefined">
        <div class="md-layout">
            <div class="md-layout-item md-size-30">
              <h4 style="font-weight:bold">Question</h4>
            </div>
            <div class="md-layout-item md-size-15">
              <h4 style="font-weight:bold">Source</h4>
            </div>
            <div class="md-layout-item md-size-30">
               <h4 style="font-weight:bold">Intent</h4>
            </div>
            <div class="md-layout-item md-size-6">
                <h4 style="font-weight:bold">Edit</h4>
            </div>
            <div class="md-layout-item md-size-6">
                <h4 style="font-weight:bold">Delete</h4>
            </div>
        </div>
        <div class="md-layout" v-for="(unanswered, index) in unanswered_list" :key="index">
            <div class="md-layout-item md-size-30">
                <p v-if="!unanswered_list[index].edit">{{ unanswered_list[index].question }}</p>
                <md-field v-if="unanswered_list[index].edit">
                <md-input v-model="unanswered_list[index].question"></md-input>
              </md-field>
            </div>
            <div class="md-layout-item md-size-15">
              <p>{{unanswered_list[index].source}}</p>
            </div>
            <div class="md-layout-item md-size-30" @click="intents_dropdown">
                <md-field>
                <label>Add your intent here.....</label>
                <md-select v-model="unanswered_list[index].intent">
                    <md-option v-for="(intent, index) in intent_list" :key="index"
                    :value="intent">{{ intent }}</md-option>
                </md-select>
              </md-field>
            </div>
            <div class="md-layout-item md-size-6">
                <md-button class="md-just-icon md-warning md-simple show-button" @click="edit_item(index)">
      <md-icon>edit</md-icon>
    </md-button>
            </div>
            <div class="md-layout-item md-size-6">
               <md-button class="md-just-icon md-danger md-simple show-button" @click="delete_item(index)">
      <md-icon>close</md-icon>
    </md-button>
            </div>
        </div>
        <md-button class="md-success" style="float: right" @click="add_questions">Save</md-button>
            </md-card-content>
        </md-card>
    </div>
</template>
 
<script>
import axios from 'axios'
import swal from 'sweetalert2'
export default {
data () {
  return {
      unanswered_url: process.env.GET_UNANSWERED_URL,
      unanswered_list: [],
      user_role: this.$session.get('UserInformation').role,
      companyname: this.$session.get('UserInformation').company_name,
      companyid: this.$session.get('UserInformation').company_id,
      intent_list: [],
}
},
created () {
    if(this.user_role=="SuperAdmin") {
    if(this.$session.has('CompanyId')) {
    this.companyid = this.$session.get('CompanyId')
    this.companyname = this.$session.get('CompanyName')
  }
  else {
    this.companyid = this.$session.get('UserInformation').company_id
    this.companyname = this.$session.get('UserInformation').company_name
  }
  }
  else if(this.user_role == "Admin") {
    this.companyid = this.$session.get('UserInformation').company_id
    this.companyname = this.$session.get('UserInformation').company_name
  }
    if(this.companyname!==undefined) {
  axios.post(this.unanswered_url, {
    companyname: this.companyname,
    companyid: this.companyid,
    // role: this.$session.get('UserInformation').role,
    // superadmincompanyname: this.$session.get('UserInformation').company_name,
    // superadmincompanyid: this.$session.get('UserInformation').company_id,
    email: this.$session.get('UserInformation').email,
    isDelete: "False",
    licensekey: this.$session.get('UserInformation').license_key,
    token: this.$session.get('UserInformation').tokens
  }).then(response => {
    var data = response.data
    for (var i in data[0]){
        this.unanswered_list.push({
            question: data[0][i],
            intent: '',
            source: data[1][i],
            edit: false })
    }
  })
  .catch(error => {
      this.$toastr.e(error.message)
  }) 
    }
},
methods: {
    delete_item (index) {
        axios.post(this.unanswered_url, {
             companyid: this.companyid,
             companyname: this.companyname, 
             question: this.unanswered_list[index].question,
             source: this.unanswered_list[index].source,
             isDelete: "True",
            //  superadmincompanyname: this.$session.get('UserInformation').company_name,
            //  superadmincompanyid: this.$session.get('UserInformation').company_id,
            //  role: this.$session.get('UserInformation').role,
            email: this.$session.get('UserInformation').email,
             licensekey: this.$session.get('UserInformation').license_key,
             token: this.$session.get('UserInformation').tokens
         })
         .then(response => {
             if(response.data=="deleted") {
             this.unanswered_list.splice(index,1)
             }
         })
         .catch(error => {
             this.$toastr.e(error.message)
         })
    },
    edit_item (index) {
      this.unanswered_list[index].edit = !(this.unanswered_list[index].edit)
    },
    add_questions () {
        var questionlist = []
        var temp_list = []
        var check = true
        for(var i in this.unanswered_list) {
            if(this.unanswered_list[i].intent=="") {
                swal({
                    text: "Please enter all the intent fields",
                    type: 'warning'
                })
                check = false
            }
        }
        for(var i in this.unanswered_list) {
            temp_list.push(this.unanswered_list[i].question, this.unanswered_list[i].intent)
            questionlist.push(temp_list)
            temp_list = []
        }
        if(check==true) {
        axios.post(this.unanswered_url, {
            companyname: this.companyname,
            companyid: this.companyid,
            isAdd: "True",
            questionintent: questionlist,
            // role: this.$session.get('UserInformation').role,
            // superadmincompanyname: this.$session.get('UserInformation').company_name,
            // superadmincompanyid: this.$session.get('UserInformation').company_id,
            email: this.$session.get('UserInformation').email,
            licensekey: this.$session.get('UserInformation').license_key,
            token: this.$session.get('UserInformation').tokens
        }).then(response => {
            console.log(response.data)
            if(response.data=="Added successfully") {
                swal({
                    text: "Added successfully",
                    type: "success"
                })
            }
        })
    }
    },
    intents_dropdown () {
        axios.post(this.unanswered_url, {
            companyid: this.companyid,
            companyname: this.companyname,
            isDropdown: "True",
            // role: this.$session.get('UserInformation').role,
            // superadmincompanyname: this.$session.get('UserInformation').company_name,
            // superadmincompanyid: this.$session.get('UserInformation').company_id,
            email: this.$session.get('UserInformation').email,
            licensekey: this.$session.get('UserInformation').license_key,
            token: this.$session.get('UserInformation').tokens
        })
        .then(response => {
           this.intent_list = response.data
        })
    },
  write(route) {
    this.render_component = route.page
  }
}
}
</script>
