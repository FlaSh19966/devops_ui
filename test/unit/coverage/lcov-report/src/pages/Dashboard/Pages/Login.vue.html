import DashboardLayout from '@/pages/Dashboard/Layout/DashboardLayout.vue'
import {LoginCard} from '@/components'
import {SweetAlert} from '@/pages'
import axios from 'axios'
import swal from 'sweetalert2'
import sidebar_items from './sidebar_items'
 
export default {
  components: {
    LoginCard,
    SweetAlert,
    DashboardLayout,
  },
  beforeCreate () {
    if(this.$session.exists()) {
      this.$session.destroy()
    }
  },
  data () {
    return {
      user: {
      email: '',
      password: '',
      },
      url: process.env.LOGIN_URL,
      validate_url: process.env.VALIDATE_URL,
      forgot_password_url: process.env.FORGOT_PASSWORD_LINK_URL,
      validation_c_id: /=([0-9a-zA-Z]+)&t/g,
      validation_token: /token=([0-9a-zA-Z._-]+)$/g,
    }
  },
  methods: {
    create () {
      if(this.user.name=='' || this.user.password==''){
        swal({
                title: "<h4>Kindly enter the email and password</h4>"
            });
      }
      else {           
      this.call_api(); 
      }
    },
        redir (message){
        if(message=="Validation error"){
              swal({
                title: "<h4>Your account has not been validated for login.<br>Please visit your e-mail</h4>",
                type: "error"
                })
            }
            else if(message=="Status Not Approve") {
          swal({
            type: "warning",
            text: "Your registration has not been approved or it has been declined. Kindly check your mail"
          })
        }
        else if(message=="null") {
          swal({
            type: 'error',
            text: 'Invalid Credentials!',
            showConfirmButton: false,
            timer: 1000,
              });
        }
        else {
              this.$session.start()
              this.$session.set('UserInformation', message.userinformation)
              this.update_menu_list (message.menubind.menu_bind);
             this.$router.push({path: "/dashboard"})
         }
          this.user.email = '';
          this.user.password = '' 
        },
        call_api () { 
          axios.post(this.url, this.user).then(response => {this.redir(response.data)}) 
        },
        forgot_password()
         {
var vm = this;
swal({
  title: 'Enter your email address',
  input: 'email',
  inputPlaceholder: 'Enter your email address here....',
  showCloseButton: true
})
.then(
  result => {
    if(result.value){
      vm.email = result.value
      axios.post(vm.forgot_password_url, {
        email: vm.email
      })
      .then(response => {
        if(response.data=="Email Error"){
          swal({
            text: "This E-Mail does not exist",
            type: "error"
          })
        }
        else if(response.data=="mail sent"){
          swal({
            text: "We have sent a link to reset your password. Kindly check your e-mail"
          })
      }
       })
    }
  }
);
  },
  
  update_menu_list (list) {
    var menu = []
    for(var i in sidebar_items) {
      if(list.includes(sidebar_items[i].name)){
        menu.push(sidebar_items[i])
      }
    }
    this.$session.set('UserMenu', menu)
  }
},
  beforeMount() {
    var link = window.location.href;
    var company_id, token_id;
 
 
    company_id = link.match(this.validation_c_id);
    company_id = String(company_id);
    company_id = company_id.substring(1,company_id.length-2)
 
 
    token_id = link.match(this.validation_token);
    token_id = String(token_id);
    token_id = token_id.substring(6,token_id.length);
 
if(company_id!='' && token_id!=''){
    axios.post(this.validate_url, 
    {
    company_id: company_id,
    token: token_id
    })
     .then(response => {
      if(response.data=='already verify'){
        swal({text:"This Email verification link is Expired. You already Verified your account. Kindly login"});
        window.location = link.substring(0,link.indexOf("?"))
      }
      else if(response.data=='Email verification Successfully'){
        swal({
          type: 'success',
          text:"Your email verification is successfully done . Please use Email id as your username and password for login"});
          window.location = link.substring(0,link.indexOf("?"))
            }
            else if(response.data=="User Doesn't Exist"){
              swal({text:"User Doesn't Exist"});
              window.location = link.substring(0,link.indexOf("?"))
            }
    })
    }
  }
}
