<template>
     <div class="md-layout text-center">
    <div class="md-layout-item md-size-33 md-medium-size-50 md-small-size-70 md-xsmall-size-100">
      <login-card header-color="green">
        <h4 slot="title" class="title" style="margin-top:25px">Forgot Password</h4>
                <md-field class="md-form-group" slot="inputs">
          <md-icon>lock_outline</md-icon>
          <label>New Password...</label>
          <md-input v-model="new_password" type="password"></md-input>
        </md-field>
    <span style="color:#FF0000;" v-if="validPassword!=true && new_password!=''" slot="inputs">{{ validPassword }}</span>
 
        <md-field class="md-form-group" slot="inputs">
          <md-icon>lock_outline</md-icon>
          <label>Confirm New Password...</label>
          <md-input type="password" @keyup.enter="create()" v-model="confirm_password"></md-input>
        </md-field>
    <span style="color:#FF0000;" v-if="validConfirmPassword!=true && confirm_password!=''" slot="inputs">{{ validConfirmPassword }}</span>        
        <md-button slot="footer" class="md-simple md-success md-lg" @click="create()">
          Submit New Password
        </md-button>
      </login-card>
    </div>
    </div>
</template>
 
<script>
 
import swal from 'sweetalert2'
import {LoginCard} from '@/components'
import axios from 'axios'
export default {
    components: {
        LoginCard,
    },
    data() {
        return {
        new_password_url: process.env.FORGOT_PASSWORD_UPDATE_URL,
        new_password: '',
        confirm_password: '',
        reg_pwd: /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[$@$!%*?&])[A-Za-z\d$@$!%*?&]{8,}/,
        token_regex: /token=([0-9a-zA-Z._-]+)/g
    }
    },
    computed: {
        validPassword () {
        if((this.reg_pwd).test(this.new_password)) {
          return true;
        }
        else {
          return 'Password must contain minimum 8 characters, atleast 1 UpperCase Alphabet, 1 LowerCase Alphabet, 1 Number and 1 Special Character'
        }
      },
       validConfirmPassword () {
        if((this.reg_pwd).test(this.confirm_password)) {
          if(this.confirm_password==this.new_password){
          return true;
          }
        }
        else{
            return "Passwords do not match"
          }
      }
    },
    methods: {
        create() {
          if(this.validPassword==true && this.validConfirmPassword==true){
        var link = window.location.href;
        var token_id;
        token_id = link.match(this.token_regex);
        token_id = String(token_id);
        token_id = token_id.substring(6,token_id.length);
 
        if(this.validPassword==true) {
           this.call_api(token_id);
        }
          }
        },
 
        call_api(token_id) {
          axios.post(this.new_password_url, {
            token: token_id,
            password: this.new_password
          }).then(response => {
            var link = window.location.href
            if(response.data=="link expired"){
              swal({
                text: "Sorry, your link has expired"
              })
              window.location = link.substring(0,link.indexOf("?"))
            }
            else if(response.data=="Password updated"){
              swal({
                text: "Your password has been updated successfully. Now you can login with the new password"
              })
              window.location = link.substring(0,link.indexOf("?"))
            }
            else if(response.data=="User Doesn't Exist"){
              swal({
                text: "This user does not exist"
              })
              window.location = link.substring(0,link.indexOf("?"))
            }
          });
          this.new_password = '';
          this.confirm_password = ''
        }
    }
}
</script>
