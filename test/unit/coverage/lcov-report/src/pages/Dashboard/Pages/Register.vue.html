<template>
  <div class="md-layout">
 
    <div class="md-layout-item">
      <signup-card header-color="green">
        <h3 slot="title" class="title" style="color:#FFFFFF;margin-top:25px">Register</h3>
 
        <div class="md-layout-item md-size-50 md-medium-size-50 md-small-size-100 ml-auto mr-auto" 
        slot="content-left" style="min-width: 80%;max-width: 100%;">
          
          <md-field class="md-form-group">       
            <md-icon>business</md-icon>
            <label>Company Name.....</label>
            <md-input v-model="user.company_name" type="text"></md-input>
          </md-field>
          <div @click="domain_api()">
          <md-field class="md-form-group">       
            <md-icon>business</md-icon>
            
            <label for="company">Domain Name</label>
          <md-select v-model="user.domain" style="margin-left: 12px;">
            <md-option v-for="(company, index) in companies" :key="index" 
            :value="company">{{ company }}</md-option>
          </md-select>
          
            <!-- <div style="width: -webkit-fill-available;" @click="domain_api()">
            <v-select :options="companies" style="width: -webkit-fill-available;margin-left:5px;" placeholder="Domain Name" v-model="user.domain">
            </v-select></div> -->
          </md-field>
          </div>
          <md-field class="md-form-group">
            <md-icon>vpn_key</md-icon>
            <label>License Key</label>
                    <md-input v-model="license_key" disabled></md-input>
          </md-field>
 
          <md-field class="md-form-group">
            <md-icon>email</md-icon>
            <label>Email...</label>
                    <md-input v-model="user.email" type="text"></md-input>
          </md-field> 
 
          <md-field class="md-form-group">
            <md-icon>lock_outline</md-icon>
            <label>Password....</label>
            <md-input v-model="user.password" type="password"  @keyup.enter="create()"></md-input>
          </md-field>
          <span style="color:#FF0000;" v-if="validPassword!=true && user.password!=''">{{ validPassword }}</span>
          
          <div class="button-container">
            <md-button href class="md-success md-round mt-4" slot="footer" @click="create()">Register</md-button>
          </div>
 
        </div>
      </signup-card>
    </div>
  </div>
</template>
<script>
  import {SignupCard} from '@/components'
  import {LoginCard} from '@/components'
  import swal from 'sweetalert2'
  import VeeValidate from 'vee-validate'
  import axios from 'axios'
  export default {
    components: {
      SignupCard,
      VeeValidate,
      LoginCard,
    },
    data () {
      return {
        license_key: Math.random().toString(36).substring(2, 15) + Math.random().toString(36).substring(2, 15) + "  (Trial Key)",
        user: {
        company_name: '',
        email: '',
        password: '',
        domain: '',
        },
        companies: [],
        domain_url: process.env.DOMAIN_NAME_URL,
        url: process.env.REGISTER_URL,
        reg_pwd: /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[$@$!%*?&])[A-Za-z\d$@$!%*?&]{8,}/,
        reg_email: /^([0-9a-zA-Z]([-.\w]*[0-9a-zA-Z])*@([0-9a-zA-Z][-\w]*[0-9a-zA-Z]\.)+[a-zA-Z]{2,9})$/,
        reg_error: "E11000",
        unique: false,
      }
    },
    computed: {
      validEmail () {
        if((this.reg_email).test(this.user.email)) {
          return true;
        }
        else {
          return false
        }
      },
      valid_cname () {
        if((this.user.company_name)!='') {
          return true;
        }
        else {
          return false;
          }
        },
      validPassword () {
        if((this.reg_pwd).test(this.user.password)) {
          return true;
        }
        else {
          return 'Password must contain minimum 8 characters, atleast 1 UpperCase Alphabet, 1 LowerCase Alphabet, 1 Number and 1 Special Character'
        }
      },
    },
    methods: { 
      domain_api() {
        axios.get(this.domain_url).then(response => {
          this.companies = []
          for(var i in response.data){
            this.companies.push(response.data[i])
          }
        })
      },
      create () {
        if((this.validEmail) && (this.validPassword==true) && (this.valid_cname)) {
          this.call_api();
        }
        else if(!(this.valid_cname)){
          swal({ 
            type: 'error',
            position: 'top-right',
            text: 'Company Name is required',
            showConfirmButton: false,
            timer: 1000,
            });
          }
          else if(!(this.validEmail)){
            swal({ 
            type: 'error',
                        text: 'Please enter a valid email',
                        showConfirmButton: false,
                        timer: 1000,
                        });
          }
          else if((this.user.password=='')){
            swal({ 
            type: 'error',
                        text: 'Password field is blank',
                        showConfirmButton: false,
                        timer: 1000,
                        });
          }
        },
      call_api () {
         axios.post(this.url, this.user).then(response => 
         {
        if((response.data) == "Registered Successfully") {
          swal({
          position: 'top-center',
          width: '100px',
          title: '<h4>Account created successfully!!<br>Kindly check your e-mail for account validation</h4>',
          type: "success"
        });
        }
        else if((response.data).includes("This email is duplicate")){
        swal({ 
            type: 'error',
            text: response.data,
            });
        }
        else if((response.data).includes("unsuccessful registeration")){
        swal({ 
            type: 'error',
            text: "Your registration was unsuccessful. Please try again.",
            });
        }
        else {
        swal({ 
            type: 'error',
            text: "Cannot register your account",
            showConfirmButton: false,
            timer: 1000
            });
        }
      this.user.email = '';
      this.user.company_name = '';
      this.user.password = '';
      this.user.domain = '';
         });
      },
 
    }
  }
 
</script>
